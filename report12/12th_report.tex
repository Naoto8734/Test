	\item 関数inorderのソースコード
		\begin{lstlisting}
void inorder(struct node *p) {
	if (p==NULL)
		return;
	inorder(p->left);
	printf("節%cに立ち寄りました\n", n2c(p->label));
	inorder(p->right);
}
		\end{lstlisting}
		\mbox{}\newline
	\item 関数postorderのソースコード
		\begin{lstlisting}
void postorder(struct node *p) {
	if (p==NULL)
		return;
	postorder(p->left);
	postorder(p->right);
	printf("節%cに立ち寄りました\n", n2c(p->label));
}
		\end{lstlisting}
